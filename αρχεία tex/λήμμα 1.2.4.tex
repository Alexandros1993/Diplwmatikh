\begin{lemma}\label{lem:3-4}Έστω $n\gr 1$ και $\beta_1,\ldots ,\beta_n$ διακεκριμένοι πραγματικοί αριθμοί.
Αν $c_0,c_1,\ldots ,c_{n-1}$ είναι οι σταθερές που ορίστηκαν στην \eqref{eq:3-3} τότε το σύστημα γραμμικών
εξισώσεων
\begin{equation}\label{eq:3-4}\sum_{j=1}^n\beta_j^kx_j=c_k,\qquad k=0,1,\ldots ,n-1\end{equation}
έχει μοναδική λύση $\varrho_1,\ldots ,\varrho_n$. Για κάθε πολυώνυμο βαθμού μικρότερου ή ίσου από $n-1$ ισχύει
$$\sum_{j=1}^nr(\beta_j)\varrho_j=\frac{1}{\sqrt{\pi }}\int_{-\infty }^{\infty }e^{-x^2}r(x)\,dx.$$
\end{lemma}


\begin{proof} Ή ύπαρξη και η μοναδικότητα της λύσης       $\varrho_1,\dots,\varrho_n$ προκύπτει άμεσα από το γεγονός ότι η ορίζουσα του συστήματος γραμμικών εξισώσεων
\begin{align*}
x_1+x_2+\dots+x_n &= c_0\\
\beta_1x_1+\beta_2x_2+\dots+\beta_nx_n &= c_1\\
\beta_1^2x_1+\beta_2^2x_2+\dots+\beta_n^2x_n &= c_2\\ \vdots \\
\beta_1^{n-1}x_1+\beta_2^{n-1}x_2+\dots+\beta_n^{n-1}&=c_{n-1}
\end{align*}
είναι η ορίζουσα \tl{Vandermode} $$\begin{vmatrix}
1&1&\dots&1\\ \beta_1&\beta_2&\dots&\beta_n\\ \beta_1^2&\beta_2^2&\dots&\beta_n^2\\ \vdots&\hfill&\hfill&\vdots\\
\beta_1^{n-1}&\beta_2^{n-1}&\dots&\beta_n^{n-1}\\
\end{vmatrix}=\prod_{0\leq j < i \leq n} (\beta_{i} - \beta_{j})\neq0$$ 
Για το πολυώνυμο $r(x)=\sum_{k=1}^{n-1}a_kx^k$ βαθμού το πολύ $n-1$ έχουμε 
\begin{align*}
\sum_{j=1}^{n}r(\beta_j)\varrho_j &= \sum_{j=1}^{n}\sum_{k=0}^{n-1}a_k\beta_j^k\varrho_j\\ &=\sum_{k=0}^{n-1}a_k\sum_{j=1}^{n}\beta_j^k\varrho_j\\ &= \sum_{k=0}^{n-1}a_kc_k\\ &= \frac{1}{\sqrt{\pi}}\sum_{k=0}^{n-1}\int_{-\infty}^{\infty}e^{-x^2}x^kdx\\ &= \frac{1}{\sqrt{\pi}}\int_{-\infty}^{\infty}e^{-x^2}r(x)dx.\\
\end{align*}

\end{proof}
